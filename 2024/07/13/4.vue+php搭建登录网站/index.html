            el: '#app',

                return {
                    visible: false,

                        username: '',
                        password: '',

            },
            created() {
                this.login2();

            methods: {
                login: function () {
                    axios.post('/api/login.php', this.userData)
                        .then((response) => {
                            console.log(response);
                            if (response.data.includes('success')) {
                                this.open1();
                            } else {
                                this.open4();
                            }
                        })
                        .catch((error) => {
                            console.log('发送失败:', error);
                        })
                },
                open1() {
                    this.$notify({
                        title: '成功',
                        message: '这是一条成功的提示消息',
                        type: 'success'
                    });
                },
                open4() {
                    this.$notify.error({
                        title: '错误',
                        message: '这是一条错误的提示消息'
                    });
                }
            }

    解释说明

    1. 用户框和密码框里的值通过Vue.js框架中的v-model指令与下方Vue框架的里data选项里的userData对象里的username和password绑定
    2. 登录按钮标签里使用click点击事件并绑定methods选项里的login方法，使用户在输入完用户名和密码并点击登录按钮之后触发login方法

$input=json_decode($inputJSON,true);
if ($input == null) {
    header('HTTP/1.1 400 Bad Request');
    echo json_encode(['error' => 'no JSON']);
} else {
    $name = $input['username'];
    $pass = $input['password'];
    $sql = "select * from users where username = '$name' and password = '$pass'";
    $result = mysqli_query($link, $sql);

        echo "success";

        echo "fail";
    }
}
